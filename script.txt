const SHEET_NAME = "Ryan-Credit Card";
const SPREADSHEET_NAME = "RyCo budget";

// ─── Main function (runs every 5 minutes via trigger) ─────────────────────────

function checkEmails() {
  const threads = GmailApp.search('from:no-reply@discovery.bank is:unread subject:"Transaction update"');

  threads.forEach(thread => {
    const messages = thread.getMessages();
    messages.forEach(message => {
      if (message.isUnread()) {
        const body = message.getPlainBody();
        const emailDate = message.getDate();

        let transaction = null;

        if (body.includes("Card payment")) {
          transaction = parseDiscoveryEmail(body, emailDate);
        } else if (body.includes("Discovery Pay")) {
          transaction = parseDiscoveryEFT(body, emailDate);
        } else if (body.includes("Incoming payment")) {
          transaction = parseIncomingPayment(body, emailDate);
        } else {
          message.markRead();
          return;
        }

        if (transaction) {
          transaction.vendor = normalizeVendor(transaction.vendor);
          // Card payments arrive uncategorized; EFT/Income parsers set their
          // own category — never override those.
          if (!transaction.category) {
            const match = categorizeVendor(transaction.vendor, loadMappings());
            if (match) {
              transaction.category    = match.category;
              transaction.subcategory = match.subcategory;
            }
          }
          if (!isDuplicate(transaction)) {
            writeToSheet(transaction);
            Logger.log("Written: " + JSON.stringify(transaction));
          } else {
            Logger.log("Duplicate skipped: " + transaction.vendor);
          }
        }

        message.markRead();
      }
    });
  });
}

// ─── Parsers ──────────────────────────────────────────────────────────────────

function parseDiscoveryEmail(body, emailDate) {
  const date = Utilities.formatDate(emailDate, Session.getScriptTimeZone(), "yyyy-MM-dd");

  const vendorMatch = body.match(/([^\n]+?)\s+[–-]\s+(R|ZAR|USD|EUR|GBP)\s+([\d,]+\.?\d*)/i);
  if (!vendorMatch) return null;

  const vendor   = vendorMatch[1].replace(/\s+\+\d+$/, "").trim();
  const currency = vendorMatch[2].trim().toUpperCase() === "R" ? "ZAR" : vendorMatch[2].trim().toUpperCase();
  const amount   = parseFloat(vendorMatch[3].replace(",", ""));

  return { date, vendor, amount, currency, category: "", subcategory: "" };
}

function parseDiscoveryEFT(body, emailDate) {
  const date = Utilities.formatDate(emailDate, Session.getScriptTimeZone(), "yyyy-MM-dd");

  const amountMatch = body.match(/R\s+([\d,]+\.?\d*)/);
  if (!amountMatch) return null;
  const amount = parseFloat(amountMatch[1].replace(",", ""));

  const toMatch = body.match(/To\s+([^\n]+)/);
  const vendor = toMatch ? toMatch[1].trim() : "EFT Payment";

  const refMatch = body.match(/Reference:\s+([^\n]+)/i);
  const reference = refMatch ? refMatch[1].trim() : "";

  return {
    date,
    vendor,
    amount,
    currency: "ZAR",
    category: "Transfer",
    subcategory: reference || "EFT",
  };
}

function parseIncomingPayment(body, emailDate) {
  const date = Utilities.formatDate(emailDate, Session.getScriptTimeZone(), "yyyy-MM-dd");

  const amountMatch = body.match(/R\s+([\d,]+\.?\d*)/);
  if (!amountMatch) return null;
  const amount = parseFloat(amountMatch[1].replace(",", ""));

  const refMatch = body.match(/Reference:\s+([^\n]+)/i);
  const reference = refMatch ? refMatch[1].trim() : "";
  const vendor = reference || "Incoming Payment";

  const isCreditCardPayment = reference.toLowerCase().includes("credit card")
    || reference.toLowerCase().includes("cc payment")
    || reference.toLowerCase().includes("card payment");

  return {
    date,
    vendor,
    amount,
    currency: "ZAR",
    category:    isCreditCardPayment ? "Transfer" : "Income",
    subcategory: isCreditCardPayment ? "Credit Card Payment" : "Incoming Payment",
  };
}

// ─── Vendor normalization & auto-categorization ──────────────────────────────

// Strip payment-processor noise so the sheet stores a clean display name:
// "YOCO *PAULINES COFFE CAP" → "PAULINES COFFE CAP".
const PROCESSOR_PREFIXES = /^(yoco|sq|zapper|snapscan|payfast|ikhokha|paygate|karri|paypal|pos)\s*[*\-–:]?\s*/i;

function normalizeVendor(raw) {
  let v = String(raw || "").trim();
  v = v.replace(PROCESSOR_PREFIXES, "");
  v = v.replace(/^[*\-–:\s]+/, "");        // stray leading separators
  v = v.replace(/\s{2,}/g, " ").trim();
  return v || String(raw || "").trim();     // never normalize down to nothing
}

// Mapping tab: key | category | subcategory. Keys are matched as
// case-insensitive substrings of the vendor, longest key first — so short
// generic keys ("pharm", "coffee") work as keyword rules in the same table.
function loadMappings() {
  const sheet = openSpreadsheet().getSheetByName("Mapping");
  if (!sheet) return [];
  return sheet.getDataRange().getValues().slice(1)
    .filter(r => r[0] && r[1])
    .map(r => ({
      key:         String(r[0]).toLowerCase().trim(),
      category:    String(r[1]).trim(),
      subcategory: r[2] ? String(r[2]).trim() : "",
    }))
    .sort((a, b) => b.key.length - a.key.length);
}

function categorizeVendor(vendor, mappings) {
  const v = String(vendor || "").toLowerCase().replace(/\s{2,}/g, " ").trim();
  if (!v) return null;
  for (const m of mappings) {
    if (v.indexOf(m.key) !== -1) return m;
  }
  return null;
}

// Fill category/subcategory for existing rows that have none. Writes plain
// values (not formulas) into columns E/F; never touches rows that already
// have a category. Returns the number of rows updated.
function categorizeBacklog() {
  const ss = openSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) return 0;
  const mappings = loadMappings();
  const data = sheet.getDataRange().getValues();
  let updated = 0;
  for (let i = 1; i < data.length; i++) {
    const hasCategory = String(data[i][4] || "").trim();
    if (hasCategory && hasCategory.toLowerCase() !== "uncategorised") continue;
    const match = categorizeVendor(data[i][1], mappings);
    if (match) {
      sheet.getRange(i + 1, 5, 1, 2).setValues([[match.category, match.subcategory]]);
      updated++;
    }
  }
  Logger.log("categorizeBacklog updated " + updated + " rows");
  return updated;
}

// OPTIONAL, run manually with care: re-normalizes vendor names on existing
// rows. Caveat: the app keys localStorage tags and recurring-exclusions by
// vendor name, so renaming historical vendors orphans those.
function normalizeBacklog() {
  const sheet = openSpreadsheet().getSheetByName(SHEET_NAME);
  if (!sheet) return;
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    const clean = normalizeVendor(data[i][1]);
    if (clean !== data[i][1]) sheet.getRange(i + 1, 2).setValue(clean);
  }
}

// ─── Sheet helpers ────────────────────────────────────────────────────────────

function isDuplicate(transaction) {
  const files = DriveApp.getFilesByName(SPREADSHEET_NAME);
  if (!files.hasNext()) throw new Error("Spreadsheet not found");
  const ss    = SpreadsheetApp.open(files.next());
  const sheet = ss.getSheetByName(SHEET_NAME);
  const data  = sheet.getDataRange().getValues();

  return data.some(row =>
    row[0] === transaction.date &&
    row[1] === transaction.vendor &&
    row[2] === transaction.amount
  );
}

function writeToSheet(transaction) {
  const files = DriveApp.getFilesByName(SPREADSHEET_NAME);
  if (!files.hasNext()) throw new Error("Spreadsheet not found: " + SPREADSHEET_NAME);
  const ss    = SpreadsheetApp.open(files.next());
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) throw new Error("Sheet tab not found: " + SHEET_NAME);

  sheet.appendRow([
    transaction.date,
    transaction.vendor,
    transaction.amount,
    transaction.currency,
    transaction.category,
    transaction.subcategory,
  ]);
}

// ─── Web app endpoint ─────────────────────────────────────────────────────────
// GET  ?sheet=transactions|budgets|assets|settings|goals|watchlists|networth
//              |mapping|fixed|investments
//   → 2D array of the tab's values (header row first), [] if the tab is missing.
// POST body = JSON string (send WITHOUT a Content-Type header so the browser
//   uses text/plain and skips the CORS preflight Apps Script can't answer):
//   { "action": "setSetting",    "key": "envelopeMode", "value": "true" }
//   { "action": "setBudgets",    "budgets": { "Groceries": 4000, ... } }
//   { "action": "setGoals",      "goals": [{ name, target, saved, deadline, icon }, ...] }
//   { "action": "setWatchlists", "watchlists": [{ name, type, match, monthlyLimit }, ...] }
//   { "action": "setFixed",      "fixed": [{ name, amount, category, subcategory, day, active }, ...] }
//   { "action": "addMapping",    "key": "vendor substring", "category": "...", "subcategory": "..." }
//   → { "ok": true } | { "ok": false, "error": "..." }  (addMapping also returns "updated": n)
// NOTE: after editing this script you must Deploy → Manage deployments →
// edit → New version. An uncaught throw here returns an HTML error page,
// which the browser reports as a CORS failure.

const READ_TABS = {
  transactions: () => SHEET_NAME,
  budgets:      () => "Budgets",
  assets:       () => "Assets",
  settings:     () => "Settings",
  goals:        () => "Goals",
  watchlists:   () => "Watchlists",
  networth:     () => "NetWorthHistory",
  mapping:      () => "Mapping",
  fixed:        () => "Fixed",
  investments:  () => "Investments",
};

function openSpreadsheet() {
  const files = DriveApp.getFilesByName(SPREADSHEET_NAME);
  if (!files.hasNext()) throw new Error("Spreadsheet not found: " + SPREADSHEET_NAME);
  return SpreadsheetApp.open(files.next());
}

function getOrCreateSheet(ss, name, headers) {
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.appendRow(headers);
  }
  return sheet;
}

function doGet(e) {
  try {
    const tab = e.parameter.sheet;
    if (!READ_TABS[tab]) return jsonResponse({ error: "Unknown sheet: " + tab });

    const ss    = openSpreadsheet();
    const sheet = ss.getSheetByName(READ_TABS[tab]());
    if (!sheet) return jsonResponse([]); // tab doesn't exist yet — return empty

    return jsonResponse(sheet.getDataRange().getValues());
  } catch (err) {
    return jsonResponse({ error: String(err) });
  }
}

function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    const ss = openSpreadsheet();

    switch (payload.action) {
      case "setSetting":
        upsertSetting(ss, String(payload.key), String(payload.value));
        break;
      case "setBudgets":
        rewriteRows(ss, "Budgets", ["Category", "Limit"],
          Object.keys(payload.budgets).map(k => [k, Number(payload.budgets[k]) || 0]));
        break;
      case "setGoals":
        rewriteRows(ss, "Goals", ["Name", "Target", "Saved", "Deadline", "Icon"],
          (payload.goals || []).map(g => [g.name, Number(g.target) || 0, Number(g.saved) || 0, g.deadline || "", g.icon || ""]));
        break;
      case "setWatchlists":
        rewriteRows(ss, "Watchlists", ["Name", "Type", "Match", "MonthlyLimit"],
          (payload.watchlists || []).map(w => [w.name, w.type, w.match, Number(w.monthlyLimit) || 0]));
        break;
      case "setFixed":
        rewriteRows(ss, "Fixed", ["Name", "Amount", "Category", "Subcategory", "Day", "Active"],
          (payload.fixed || []).map(f => [
            f.name, Number(f.amount) || 0, f.category || "", f.subcategory || "",
            Number(f.day) || 1, f.active ? "TRUE" : "FALSE",
          ]));
        break;
      case "addMapping": {
        // Teach loop: save the mapping, then backfill every uncategorized
        // row it matches so one tap in the app fixes past and future.
        const sheet = getOrCreateSheet(ss, "Mapping", ["Key", "Category", "Subcategory"]);
        const key = String(payload.key || "").toLowerCase().trim();
        if (!key) return jsonResponse({ ok: false, error: "Empty mapping key" });
        const exists = sheet.getDataRange().getValues().slice(1)
          .some(r => String(r[0]).toLowerCase().trim() === key);
        if (!exists) sheet.appendRow([key, payload.category || "", payload.subcategory || ""]);
        const updated = categorizeBacklog();
        return jsonResponse({ ok: true, updated });
      }
      default:
        return jsonResponse({ ok: false, error: "Unknown action: " + payload.action });
    }
    return jsonResponse({ ok: true });
  } catch (err) {
    return jsonResponse({ ok: false, error: String(err) });
  }
}

function upsertSetting(ss, key, value) {
  const sheet = getOrCreateSheet(ss, "Settings", ["Key", "Value"]);
  const data  = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === key) {
      sheet.getRange(i + 1, 2).setValue(value);
      return;
    }
  }
  sheet.appendRow([key, value]);
}

// Rewrites all data rows below the header. Destructive by design — the app
// sends the complete list each save (fine for a single user).
function rewriteRows(ss, name, headers, rows) {
  const sheet = getOrCreateSheet(ss, name, headers);
  const last  = sheet.getLastRow();
  if (last > 1) sheet.getRange(2, 1, last - 1, sheet.getLastColumn()).clearContent();
  if (rows.length) sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
}

// ─── Net worth snapshot (run monthly via time-driven trigger) ─────────────────

function snapshotNetWorth() {
  const ss    = openSpreadsheet();
  const asset = ss.getSheetByName("Assets");
  if (!asset) return;

  const rows = asset.getDataRange().getValues().slice(1);
  let assets = 0, liabilities = 0;
  rows.forEach(r => {
    const value = Number(r[1]) || 0;
    if (String(r[2]).toLowerCase() === "liability") liabilities += value;
    else if (r[0]) assets += value;
  });

  // Investments count as assets (Value is column E of the Investments tab)
  const inv = ss.getSheetByName("Investments");
  if (inv) {
    inv.getDataRange().getValues().slice(1).forEach(r => {
      if (r[0] || r[1]) assets += Number(r[4]) || 0;
    });
  }

  const history = getOrCreateSheet(ss, "NetWorthHistory", ["Date", "Assets", "Liabilities", "Net"]);
  const date = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd");
  history.appendRow([date, assets, liabilities, assets - liabilities]);
}

// Run once from the editor to install the monthly snapshot trigger.
function setupTriggers() {
  const exists = ScriptApp.getProjectTriggers()
    .some(t => t.getHandlerFunction() === "snapshotNetWorth");
  if (!exists) {
    ScriptApp.newTrigger("snapshotNetWorth")
      .timeBased().onMonthDay(1).atHour(6)
      .create();
  }
}

function jsonResponse(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
